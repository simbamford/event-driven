78  bool check(int x, int y, int p, int ts)
79  {
80 
81  auto add = true;
82 
83  if(x_tfilter) {
84  if(p == POL(x, y)) {
85  int dt = ts - SAE(x, y);
86  if(dt < 0)
87  dt += vtsHelper::max_stamp;
88  if(dt < t_tfilter) {
89  SAE(x, y) = ts;
90  return false;
91  }
92  }
93  }
94 
95  POL(x, y) = p;
96  SAE(x, y) = ts;
97 
98  if(x_sfilter) {
99  add = false;
100  auto xl = std::max(x-s_sfilter, 0);
101  auto xh = std::min(x+s_sfilter+1, (int)(res.width)); //+1 becuase I use < sign
102  auto yl = std::max(y-s_sfilter, 0);
103  auto yh = std::min(y+s_sfilter+1, (int)(res.height)); //+1 becuase I use < sign
104 
105  for(auto xi = xl; xi < xh; ++xi) {
106  for(auto yi = yl; yi < yh; ++yi) {
107  int dt = ts - SAE(xi, yi);
108  if(dt < 0) {
109  dt += vtsHelper::max_stamp;
110  SAE(xi, yi) -= vtsHelper::max_stamp;
111  }
112  if(dt && dt < t_sfilter) {
113  add = true;
114  break;
115  }
116  }
117  }
118  }
119 
120  return add;
121  }
